function handleGlobalMouseMove(event) {

    const avgMousePosition = new THREE.Vector2(0, 0);

    avgMousePosition.divideScalar(mouseHistory.length);

    const startTime = performance.now();
    
    function animateCompletion(currentTime) {
        if (!rotationGroup) return;
        const elapsed = currentTime - startTime;
        const progress = Math.min(elapsed / duration, 1);
        const angle = currentAngle + (targetAngle - currentAngle) * progress;
        
        rotationGroup.rotation.set(0, 0, 0);
        rotationGroup.rotateOnAxis(rotationAxis, angle);
        
        if (progress < 1) {
            requestAnimationFrame(animateCompletion);
        } else {
            finishRotation();
        }
    }
    
    requestAnimationFrame(animateCompletion);

function getCubesInLayer(normal, clickedObject) {
    const layerCubes = [];
    const threshold = 0.9;

    const clickedPos = new THREE.Vector3();
    clickedObject.getWorldPosition(clickedPos);

    const axis = Math.abs(normal.x) > threshold ? 'x' :
                 Math.abs(normal.y) > threshold ? 'y' : 'z';
    
    // –û–ø—Ä–µ–¥–µ–ª—è–µ–º –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç—É —Å–ª–æ—è
    let layerCoord = Math.round(clickedPos[axis]);
    
    // –°–æ–±–∏—Ä–∞–µ–º –≤—Å–µ –∫—É–±–∏–∫–∏ –≤ —Å–ª–æ–µ
    Objects.forEach(cube => {
        const cubePos = new THREE.Vector3();
        cube.getWorldPosition(cubePos);
        if (Math.abs(cubePos[axis] - layerCoord) < 0.1){
            layerCubes.push(cube)
        }       
    });
    
    console.log(`–°–ª–æ–π –ø–æ –æ—Å–∏: ${axis}, –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞: ${layerCoord}, –∫—É–±–∏–∫–æ–≤: ${layerCubes.length}`)
    return { cubes: layerCubes };
}

function rotateLayer(object, normal) {
    if (isRotating || !object.parent) return;
    console.log('–í—Ä–∞—â–µ–Ω–∏–µüîÉ: ', {
        object: object.name,
        normal: { x: normal.x, y: normal.y, z: normal.z },
        camMode: CurrentActiveCam,
        CubesInLayer: cubesToRotate.length
    });

    const layerData = getCubesInLayer(normal, object);
    cubesToRotate = layerData.cubes;

    if (cubesToRotate.length === 0) return;

    // –£–¥–∞–ª—è–µ–º –ø—Ä–µ–¥—ã–¥—É—â–∏–π arrowHelper, –µ—Å–ª–∏ –æ–Ω –µ—Å—Ç—å
    if (arrowHelper) {
        scene.remove(arrowHelper);
        arrowHelper = null;
    }

    // –£–¥–∞–ª—è–µ–º –ø—Ä–µ–¥—ã–¥—É—â–∏–µ —Å—Ç—Ä–µ–ª–∫–∏ –ø—Ä–æ–≥—Ä–µ—Å—Å–∞
    progressArrows.forEach(arrow => scene.remove(arrow));
    progressArrows = [];

    // –°–æ–∑–¥–∞—ë–º –≥—Ä—É–ø–ø—É –¥–ª—è –≤—Ä–∞—â–µ–Ω–∏—è
    rotationGroup = new THREE.Group();
    const centerPoint = new THREE.Vector3();

    cubesToRotate.forEach(cube => {
        const pos = new THREE.Vector3();
        cube.getWorldPosition(pos);
        centerPoint.add(pos);
    });
    centerPoint.divideScalar(cubesToRotate.length);

    rotationGroup.position.copy(centerPoint);
    scene.add(rotationGroup);

    cubesToRotate.forEach(cube => {
        const pos = new THREE.Vector3();
        cube.getWorldPosition(pos);
        cube.position.copy(pos.sub(centerPoint));
        scene.remove(cube);
        rotationGroup.add(cube);
    });

    rotationAxis.copy(normal).normalize();
    isRotating = true;

    // –°–æ–∑–¥–∞—ë–º –Ω–∞—á–∞–ª—å–Ω—ã–µ —Å—Ç—Ä–µ–ª–∫–∏ –ø—Ä–æ–≥—Ä–µ—Å—Å–∞
    updateProgressArrows(0);

    // –°–æ–∑–¥–∞—ë–º –Ω–æ–≤—ã–π arrowHelper
    arrowHelper = new THREE.ArrowHelper(
        rotationAxis,
        rotationGroup.position,
        2,
        0xff0000
    );
    scene.add(arrowHelper);
}

function finishRotation() {
    if (!rotationGroup) return;

    cubesToRotate.forEach(cube => {
        cube.traverse(child => {
            if (child.isMesh && originalMaterials.has(child.uuid)) {
                // –í–æ—Å—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º –º–∞—Ç–µ—Ä–∏–∞–ª –∏–∑ —Ö—Ä–∞–Ω–∏–ª–∏—â–∞
                child.material = originalMaterials.get(child.uuid).clone();
                
                // –û–±–Ω–æ–≤–ª—è–µ–º —Å–≤–æ–π—Å—Ç–≤–∞
                child.material.needsUpdate = true;
                child.geometry.computeVertexNormals();
                
                // –î–ª—è –º–∞—Ç–µ—Ä–∏–∞–ª–æ–≤ —Å —ç–º–∏—Å—Å–∏–µ–π
                if (child.material.emissive) {
                    child.material.emissiveIntensity = 0.5; // –ó–Ω–∞—á–µ–Ω–∏–µ –ø–æ —É–º–æ–ª—á–∞–Ω–∏—é
                }
            }
        });
    });

    const tempContainer = new THREE.Group();
    scene.add(tempContainer);
    tempContainer.position.copy(rotationGroup.position);
    tempContainer.quaternion.copy(rotationGroup.quaternion);

    while (rotationGroup.children.length > 0) {
        const cube = rotationGroup.children[0];
        const originalPos = new THREE.Vector3().copy(cube.position);

        rotationGroup.remove(cube);
        tempContainer.add(cube);
        cube.position.copy(originalPos);
    }

    while (tempContainer.children.length > 0) {
        const cube = tempContainer.children[0];
        const worldPos = new THREE.Vector3();
        cube.getWorldPosition(worldPos);
        const worldQuater = cube.getWorldQuaternion(new THREE.Quaternion())

        // –û–∫—Ä—É–≥–ª—è–µ–º –ø–æ–∑–∏—Ü–∏–∏ –∏ –∫–≤–∞—Ç–µ—Ä–Ω–∏–æ–Ω—ã
        worldPos.x = Math.round(worldPos.x * 1000) / 1000;
        worldPos.y = Math.round(worldPos.y * 1000) / 1000;
        worldPos.z = Math.round(worldPos.z * 1000) / 1000;
        worldQuater.x = Math.round(worldQuater.x * 1000) / 1000
        worldQuater.y = Math.round(worldQuater.y * 1000) / 1000
        worldQuater.z = Math.round(worldQuater.z * 1000) / 1000
        worldQuater.w = Math.round(worldQuater.w * 1000) / 1000

        tempContainer.remove(cube);
        scene.attach(cube);
        cube.position.copy(worldPos);
        cube.quaternion.copy(worldQuater);
    }

    scene.remove(tempContainer);
    if (arrowHelper) {
        scene.remove(arrowHelper);
        arrowHelper = null;
    }
    // –£–¥–∞–ª—è–µ–º —Å—Ç—Ä–µ–ª–∫–∏ –ø—Ä–æ–≥—Ä–µ—Å—Å–∞
    progressArrows.forEach(arrow => scene.remove(arrow));
    progressArrows = [];
    scene.remove(rotationGroup);
    isRotating = false;
    cubesToRotate = [];
    rotationGroup = null;
}

/**
 * –°–æ–∑–¥–∞–µ—Ç –Ω–µ–≤–∏–¥–∏–º—ã–µ —Ç—Ä–∏–≥–≥–µ—Ä–Ω—ã–µ –∑–æ–Ω—ã –≤–æ–∫—Ä—É–≥ –∫—É–±–∏–∫–∞ –¥–ª—è –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏—è –≤—Ä–∞—â–µ–Ω–∏–π –ø–æ —Å–≤–∞–π–ø—É –Ω–∞ —Ä–µ–±—Ä–∞—Ö.

 * @returns {THREE.Mesh[]} –ú–∞—Å—Å–∏–≤ –º–µ—à–µ–π —Ç—Ä–∏–≥–≥–µ—Ä–Ω—ã—Ö –∑–æ–Ω.

export function createTriggerZones(cubeSize = 5.7) { // –£—Å—Ç–∞–Ω–∞–≤–ª–∏–≤–∞–µ–º —Ä–∞–∑–º–µ—Ä –ø–æ —É–º–æ–ª—á–∞–Ω–∏—é 6.12
    const zones = [];
    const halfSize = cubeSize / 2; // –ü–æ–ª–æ–≤–∏–Ω–∞ —Ä–∞–∑–º–µ—Ä–∞ –∫—É–±–∏–∫–∞ (3.06)
    const smallCubeSize = cubeSize / 3; // –†–∞–∑–º–µ—Ä –º–∞–ª–µ–Ω—å–∫–æ–≥–æ –∫—É–±–∏–∫–∞ (2.04)

    // –ü–æ–∑–∏—Ü–∏–∏ —Ü–µ–Ω—Ç—Ä–æ–≤ —Å–ª–æ–µ–≤ –≤–¥–æ–ª—å –∫–∞–∂–¥–æ–π –æ—Å–∏
    // –î–ª—è –∫—É–±–∏–∫–∞ 3x3x3 —Ä–∞–∑–º–µ—Ä–æ–º 6.12, —Ü–µ–Ω—Ç—Ä–∏—Ä–æ–≤–∞–Ω–Ω–æ–≥–æ –≤ (0,0,0)
    const layerPositions = [smallCubeSize, 0, -smallCubeSize]; // [2.04, 0, -2.04]

    // –†–∞–∑–º–µ—Ä—ã –∑–æ–Ω—ã. –ú–æ–∂–Ω–æ —Å–¥–µ–ª–∞—Ç—å –∏—Ö –Ω–µ–º–Ω–æ–≥–æ –±–æ–ª—å—à–µ —Ä–∞–∑–º–µ—Ä–∞ –≥—Ä–∞–Ω–∏ –º–∞–ª–µ–Ω—å–∫–æ–≥–æ –∫—É–±–∏–∫–∞.
    const zoneSize = smallCubeSize * 0.5; // –†–∞–∑–º–µ—Ä –∑–æ–Ω—ã –≤–¥–æ–ª—å —Ä–µ–±—Ä–∞ –∏ –ø–µ—Ä–ø–µ–Ω–¥–∏–∫—É–ª—è—Ä–Ω–æ –≤–Ω—É—Ç—Ä—å (–Ω–∞–ø—Ä–∏–º–µ—Ä, 2.04 * 1.1 = ~2.24)
    const zoneThickness = smallCubeSize * 0.1; // –¢–æ–ª—â–∏–Ω–∞ –∑–æ–Ω—ã (–Ω–∞—Å–∫–æ–ª—å–∫–æ –æ–Ω–∞ –≤—ã—Å—Ç—É–ø–∞–µ—Ç –Ω–∞—Ä—É–∂—É) (–Ω–∞–ø—Ä–∏–º–µ—Ä, 2.04 * 0.1 = ~0.2)
    const outwardOffset = zoneThickness / 2; // –°–º–µ—â–µ–Ω–∏–µ –æ—Ç –≥—Ä–∞–Ω–∏/—Ä–µ–±—Ä–∞ –Ω–∞—Ä—É–∂—É (—Ä–∞–≤–Ω–æ –ø–æ–ª–æ–≤–∏–Ω–µ —Ç–æ–ª—â–∏–Ω—ã –∑–æ–Ω—ã)

    // –ú–∞—Ç–µ—Ä–∏–∞–ª –¥–ª—è –∑–æ–Ω (–¥–µ–ª–∞–µ–º –µ–≥–æ –Ω–µ–≤–∏–¥–∏–º—ã–º –≤ —Ñ–∏–Ω–∞–ª—å–Ω–æ–π –≤–µ—Ä—Å–∏–∏)
    const zoneMaterial = new THREE.MeshBasicMaterial({
        color: 0x00ffff, // –¶–≤–µ—Ç –¥–ª—è –æ—Ç–ª–∞–¥–∫–∏ (–±–∏—Ä—é–∑–æ–≤—ã–π)
        wireframe: true, // –ü–æ–∫–∞–∑—ã–≤–∞–µ–º –∫–∞—Ä–∫–∞—Å –¥–ª—è –æ—Ç–ª–∞–¥–∫–∏
        transparent: true,
        opacity: 0.5, // –î–µ–ª–∞–µ–º –ø–æ–ª—É–ø—Ä–æ–∑—Ä–∞—á–Ω—ã–º –¥–ª—è –æ—Ç–ª–∞–¥–∫–∏
        visible: true // –£—Å—Ç–∞–Ω–æ–≤–∏—Ç–µ true –¥–ª—è –æ—Ç–ª–∞–¥–∫–∏, false –¥–ª—è –ø—Ä–æ–¥–∞–∫—à–µ–Ω–∞
    });

    // –û–ø—Ä–µ–¥–µ–ª—è–µ–º —Ä–µ–±—Ä–∞ –∏ —Å–≤—è–∑–∞–Ω–Ω—ã–µ —Å –Ω–∏–º–∏ —Å–≤–æ–π—Å—Ç–≤–∞
    // –î–ª—è –∫–∞–∂–¥–æ–≥–æ —Ä–µ–±—Ä–∞:
    // - fixedCoords: –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç—ã, –∫–æ—Ç–æ—Ä—ã–µ –æ—Å—Ç–∞—é—Ç—Å—è –ø–æ—Å—Ç–æ—è–Ω–Ω—ã–º–∏ –Ω–∞ —ç—Ç–æ–º —Ä–µ–±—Ä–µ (–±—É–¥—É—Ç +/- halfSize)
    // - edgeAxis: –æ—Å—å, –≤–¥–æ–ª—å –∫–æ—Ç–æ—Ä–æ–π –∏–¥–µ—Ç —Ä–µ–±—Ä–æ ('x', 'y', –∏–ª–∏ 'z')
    // - outwardDir: –≤–µ–∫—Ç–æ—Ä, —É–∫–∞–∑—ã–≤–∞—é—â–∏–π –Ω–∞—Ä—É–∂—É –æ—Ç —Ü–µ–Ω—Ç—Ä–∞ –∫—É–±–∏–∫–∞ —á–µ—Ä–µ–∑ —ç—Ç–æ —Ä–µ–±—Ä–æ
    // - triggeredFace: –≥—Ä–∞–Ω—å, –≤—Ä–∞—â–µ–Ω–∏–µ –∫–æ—Ç–æ—Ä–æ–π –∑–∞–ø—É—Å–∫–∞–µ—Ç—Å—è —ç—Ç–∏–º —Ä–µ–±—Ä–æ–º (–ø–æ —Å–≤–∞–π–ø—É –≤ –Ω–∞–ø—Ä–∞–≤–ª–µ–Ω–∏–∏ outwardDir)
    const edges = [
        // –†–µ–±—Ä–∞ –≤–¥–æ–ª—å –æ—Å–∏ Y (—Ñ–∏–∫—Å–∏—Ä–æ–≤–∞–Ω–Ω—ã–µ X, Z) - –∑–∞–ø—É—Å–∫–∞—é—Ç –≤—Ä–∞—â–µ–Ω–∏–µ R/L (–≤–æ–∫—Ä—É–≥ X)
        { fixedX: halfSize, fixedZ: halfSize, edgeAxis: 'y', outwardDir: new THREE.Vector3(1, 0, 1), triggeredFace: 'right' }, // F-R (—Å—Ç—Ä–µ–ª–∫–∏ +X -> R)
        { fixedX: halfSize, fixedZ: -halfSize, edgeAxis: 'y', outwardDir: new THREE.Vector3(1, 0, -1), triggeredFace: 'right' }, // B-R (—Å—Ç—Ä–µ–ª–∫–∏ +X -> R)
        { fixedX: -halfSize, fixedZ: halfSize, edgeAxis: 'y', outwardDir: new THREE.Vector3(-1, 0, 1), triggeredFace: 'left' }, // F-L (—Å—Ç—Ä–µ–ª–∫–∏ -X -> L)
        { fixedX: -halfSize, fixedZ: -halfSize, edgeAxis: 'y', outwardDir: new THREE.Vector3(-1, 0, -1), triggeredFace: 'left' }, // B-L (—Å—Ç—Ä–µ–ª–∫–∏ -X -> L)

        // –†–µ–±—Ä–∞ –≤–¥–æ–ª—å –æ—Å–∏ X (—Ñ–∏–∫—Å–∏—Ä–æ–≤–∞–Ω–Ω—ã–µ Y, Z) - –∑–∞–ø—É—Å–∫–∞—é—Ç –≤—Ä–∞—â–µ–Ω–∏–µ U/D (–≤–æ–∫—Ä—É–≥ Y)
        { fixedY: halfSize, fixedZ: halfSize, edgeAxis: 'x', outwardDir: new THREE.Vector3(0, 1, 1), triggeredFace: 'up' }, // F-T (—Å—Ç—Ä–µ–ª–∫–∏ +Y -> U)
        { fixedY: -halfSize, fixedZ: halfSize, edgeAxis: 'x', outwardDir: new THREE.Vector3(0, -1, 1), triggeredFace: 'down' }, // F-Btm (—Å—Ç—Ä–µ–ª–∫–∏ -Y -> D)
        { fixedY: halfSize, fixedZ: -halfSize, edgeAxis: 'x', outwardDir: new THREE.Vector3(0, 1, -1), triggeredFace: 'up' }, // B-T (—Å—Ç—Ä–µ–ª–∫–∏ +Y -> U)
        { fixedY: -halfSize, fixedZ: -halfSize, edgeAxis: 'x', outwardDir: new THREE.Vector3(0, -1, -1), triggeredFace: 'down' }, // B-Btm (—Å—Ç—Ä–µ–ª–∫–∏ -Y -> D)

        // –†–µ–±—Ä–∞ –≤–¥–æ–ª—å –æ—Å–∏ Z (—Ñ–∏–∫—Å–∏—Ä–æ–≤–∞–Ω–Ω—ã–µ X, Y) - –∑–∞–ø—É—Å–∫–∞—é—Ç –≤—Ä–∞—â–µ–Ω–∏–µ R/L (–≤–æ–∫—Ä—É–≥ X) –ø–æ –∏–∑–æ–±—Ä–∞–∂–µ–Ω–∏—é
        { fixedX: halfSize, fixedY: halfSize, edgeAxis: 'z', outwardDir: new THREE.Vector3(1, 1, 0), triggeredFace: 'right' }, // T-R (—Å—Ç—Ä–µ–ª–∫–∏ +X -> R)
        { fixedX: -halfSize, fixedY: halfSize, edgeAxis: 'z', outwardDir: new THREE.Vector3(-1, 1, 0), triggeredFace: 'left' }, // T-L (—Å—Ç—Ä–µ–ª–∫–∏ -X -> L)
        { fixedX: halfSize, fixedY: -halfSize, edgeAxis: 'z', outwardDir: new THREE.Vector3(1, -1, 0), triggeredFace: 'right' }, // Btm-R (—Å—Ç—Ä–µ–ª–∫–∏ +X -> R)
        { fixedX: -halfSize, fixedY: -halfSize, edgeAxis: 'z', outwardDir: new THREE.Vector3(-1, -1, 0), triggeredFace: 'left' }, // Btm-L (—Å—Ç—Ä–µ–ª–∫–∏ -X -> L)
    ];

    // –ì–µ–æ–º–µ—Ç—Ä–∏—è –∑–æ–Ω—ã (—à–∏—Ä–∏–Ω–∞, –≤—ã—Å–æ—Ç–∞, –≥–ª—É–±–∏–Ω–∞).
    // –®–∏—Ä–∏–Ω–∞ –∏ –≤—ã—Å–æ—Ç–∞ –ø—Ä–∏–º–µ—Ä–Ω–æ —Ä–∞–≤–Ω—ã –∑–æ–Ω–µ –≤–¥–æ–ª—å —Ä–µ–±—Ä–∞, –≥–ª—É–±–∏–Ω–∞ - —Ç–æ–ª—â–∏–Ω–∞.
    const zoneGeometry = new THREE.BoxGeometry(zoneSize, zoneSize, zoneThickness); // width, height, depth

    edges.forEach(edge => {
        const outwardVector = edge.outwardDir.clone().normalize(); // –ù–æ—Ä–º–∞–ª–∏–∑–æ–≤–∞–Ω–Ω—ã–π –≤–µ–∫—Ç–æ—Ä –Ω–∞—Ä—É–∂—É

        layerPositions.forEach((layerPos, index) => { // –ò—Å–ø–æ–ª—å–∑—É–µ–º –Ω–æ–≤—ã–µ layerPositions

            const yOffset = 5;
            // –í—ã—á–∏—Å–ª—è–µ–º –ø–æ–∑–∏—Ü–∏—é –∑–æ–Ω—ã
            const position = new THREE.Vector3();
            if (edge.edgeAxis === 'y') {
                position.set(edge.fixedX, layerPos + yOffset, edge.fixedZ);
            } else if (edge.edgeAxis === 'x') {
                position.set(layerPos, edge.fixedY + yOffset, edge.fixedZ);
            } else if (edge.edgeAxis === 'z') {
                position.set(edge.fixedX, edge.fixedY + yOffset, layerPos);
            }

            zone.lookAt(zone.position.clone().add(outwardVector));
            zone.rotateOnAxis(new THREE.Vector3(0, 1, 0), Math.PI); // –ö–æ—Ä—Ä–µ–∫—Ç–∏—Ä—É–µ–º –æ—Ä–∏–µ–Ω—Ç–∞—Ü–∏—é

            // –î–æ–±–∞–≤–ª—è–µ–º –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—å—Å–∫–∏–µ –¥–∞–Ω–Ω—ã–µ –¥–ª—è –∏–¥–µ–Ω—Ç–∏—Ñ–∏–∫–∞—Ü–∏–∏ –∑–æ–Ω—ã
            zone.userData = {
                isTriggerZone: true,
                triggeredFace: edge.triggeredFace, // 'right', 'left', 'up', 'down', 'front', 'back'
                layerIndex: index, // 0, 1, 2 (—Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É–µ—Ç layerPositions [2.04, 0, -2.04])
            };

            zones.push(zone);
        });
    });

    return zones;
